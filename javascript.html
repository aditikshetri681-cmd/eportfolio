JavaScript ES6+
// DOM Manipulation
const btn = document
  .querySelector('#addBtn');

btn.addEventListener('click', () => {
  const skills = ['HTML','CSS','JS'];
  const ul = document
    .getElementById('list');

  ul.innerHTML = skills
    .map(s => `<li>${s}</li>`)
    .join('');
});

// Async / Fetch API
async function fetchUser() {
  const res  = await
    fetch('/api/user/1');
  const data = await res.json();
  console.log(data.name);
}

// Arrow fn + destructuring
const greet = ({ name, role }) =>
  `Hello, ${name} (${role})`;

greet({
  name: 'Aditi',
  role: 'Full Stack Dev'
});
// → "Hello, Aditi (Full Stack Dev)"
